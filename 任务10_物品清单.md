### 2. 物品列表渲染（替换三卡片区域）

```javascript
function renderItemsInTab(filter) {
  _itemFilter = filter || 'all';
  switchMyTab('items');
  
  var items = _itemFilter === 'all' ? MOCK_ITEMS : MOCK_ITEMS.filter(function(i){ return i.status === _itemFilter; });
  var counts = { all:MOCK_ITEMS.length, storage:0, selling:0, auction:0, sold:0 };
  MOCK_ITEMS.forEach(function(i){ counts[i.status]++; });
  
  var h = '';
  // 筛选 chip
  h += '<div style="padding:4px 14px"><div class="my-filters" style="border:none">';
  [{k:'all',l:'全部'},{k:'storage',l:'在库'},{k:'selling',l:'在售'},{k:'auction',l:'拍卖'},{k:'sold',l:'已售'}].forEach(function(ch){
    h += '<span class="my-fchip' + (_itemFilter===ch.k?' on':'') + '" onclick="_itemFilter=\''+ch.k+'\';renderItemsInTab(_itemFilter)">' + ch.l + ' ' + counts[ch.k] + '</span>';
  });
  h += '</div></div>';
  
  // 物品列表
  h += '<div style="padding:4px 14px">';
  if (items.length === 0) {
    h += '<div style="text-align:center;padding:40px;color:#5a6e5c">暂无物品</div>';
  } else {
    var icons = {storage:'📦', selling:'🏪', auction:'🔨', sold:'📤'};
    items.forEach(function(item){
      h += '<div style="display:flex;gap:10px;align-items:center;background:#fff;border:1px solid #d0d9ce;border-radius:10px;padding:12px;margin-bottom:6px;cursor:pointer" onclick="openItemDetail(\''+item.id+'\')">';
      h += '<div style="font-size:1.5rem;width:40px;text-align:center">📦</div>';
      h += '<div style="flex:1;min-width:0"><div style="font-weight:600;font-size:.82rem">'+item.name+'</div>';
      h += '<div style="font-size:.65rem;color:#5a6e5c;margin-top:2px">'+(icons[item.status]||'')+' '+item.statusLabel()+' · '+(item.price?'¥'+item.price+' · ':'')+item.date+'</div></div>';
      h += '</div>';
    });
  }
  h += '</div>';
  
  document.getElementById('myTabItems').innerHTML = h;
}

// 辅助：status 中文
function itemStatusLabel(s) {
  return {storage:'在库', selling:'在售', auction:'拍卖中', sold:'已售'}[s] || s;
}
```

### 4. 物品详情 JS

```javascript
function openItemDetail(itemId) {
  var item = MOCK_ITEMS.find(function(i){ return i.id === itemId; });
  if (!item) return;
  
  var h = '<div style="text-align:center;padding:20px 0">';
  h += '<div style="font-size:3rem">📦</div>';
  h += '<div style="font-size:1.1rem;font-weight:700;margin:8px 0">'+item.name+'</div>';
  h += '<div style="font-size:.75rem;color:#5a6e5c">📍 '+item.location+' · '+item.date+'</div>';
  if (item.price) h += '<div style="font-size:1.2rem;font-weight:700;color:#3d6b52;margin-top:4px">¥'+item.price+'</div>';
  h += '<div style="margin-top:4px"><span style="font-size:.65rem;padding:2px 8px;border-radius:6px;font-weight:600;background:#e8f0e8;color:#3d6b52">'+itemStatusLabel(item.status)+'</span></div>';
  h += '</div>';
  
  document.getElementById('itemDetailTitle').textContent = item.name;
  document.getElementById('itemDetailBody').innerHTML = h;
  
  // 操作栏按钮
  var bar = document.getElementById('itemDetailBar');
  var btns = '';
  if (item.status === 'storage') {
    btns += '<button class="btn-pri btn-full" onclick="closeOverlay(\'overlayItemDetail\');openShelfForm(\''+item.id+'\')">🏪 上架市集</button>';
    btns += '<button class="btn-outline btn-full" onclick="closeOverlay(\'overlayItemDetail\');openAuctionForm(\''+item.id+'\')">🔨 送拍</button>';
    btns += '<button class="btn-outline btn-full" onclick="doGiftItem(\''+item.id+'\')">🎁 赠与</button>';
  } else if (item.status === 'selling') {
    btns += '<button class="btn-outline btn-full" onclick="doDelistItem(\''+item.id+'\')">⬇ 下架</button>';
  } else if (item.status === 'auction') {
    btns += '<button class="btn-outline btn-full" onclick="doCancelAuctionItem(\''+item.id+'\')">撤回送拍</button>';
  } else if (item.status === 'sold') {
    btns += '<button class="btn-pri btn-full" onclick="doRelistItem(\''+item.id+'\')">🔄 再次上架</button>';
  }
  bar.innerHTML = btns;
  bar.style.display = btns ? 'flex' : 'none';
  
  document.getElementById('overlayItemDetail').classList.add('open');
}

function closeItemDetail() {
  document.getElementById('overlayItemDetail').classList.remove('open');
  renderItemsInTab(_itemFilter);
}
```

### 5. 操作函数（纯前端，改 MOCK_ITEMS）

```javascript
function doDelistItem(id) {
  var item = MOCK_ITEMS.find(function(i){ return i.id === id; });
  if (item) { item.status = 'storage'; item.price = 0; }
  closeItemDetail();
}
function doRelistItem(id) {
  var item = MOCK_ITEMS.find(function(i){ return i.id === id; });
  if (item) { item.status = 'selling'; item.date = '刚刚上架'; }
  closeItemDetail();
}
function doCancelAuctionItem(id) {
  var item = MOCK_ITEMS.find(function(i){ return i.id === id; });
  if (item) { item.status = 'storage'; item.price = 0; }
  closeItemDetail();
}
function doGiftItem(id) {
  var phone = prompt('接收人手机号（模拟）：');
  if (!phone) return;
  var item = MOCK_ITEMS.find(function(i){ return i.id === id; });
  if (item) { item.status = 'sold'; item.buyer = phone.slice(-4); item.date = '刚刚赠出'; }
  closeItemDetail();
}
```

6. 点已售物品 → [🔄 再次上架] → 物品回到在售列表
