function goWizardStep(step) {
  if (step < 1 || step > 7) return;
  // 校验当前步
  var err = validateWizardStep(_campWizardStep);
  if (step > _campWizardStep && err) { showToast(err, 'warn'); return; }
  // 保存当前步数据
  saveWizardStep(_campWizardStep);
  _campWizardStep = step;
  renderWizardStep(step);
  // 滚动到顶部
  document.getElementById('campWizardBody').scrollTop = 0;
}

function validateWizardStep(step) {
  if (step === 1) {
    if (!_campDraft.step1.name) return '请输入营队名称';
    if (!_campDraft.step1.theme) return '请输入一句话主题';
  }
  if (step === 5) {
    if (!_campDraft.step5.builders.length) return '请至少选择一位共建人';
  }
  return null;
}

function saveWizardStep(step) {
  if (step === 1) {
    _campDraft.step1.name = document.getElementById('cwName').value.trim();
    _campDraft.step1.season = document.getElementById('cwSeason').value;
    _campDraft.step1.type = document.querySelector('input[name="cwType"]:checked')?.value || 'regular';
    _campDraft.step1.theme = document.getElementById('cwTheme').value.trim();
    _campDraft.step1.desc = document.getElementById('cwDesc').value.trim();
    _campDraft.step1.testMode = document.getElementById('cwTestMode').checked;
  }
}

function updateWizardButtons() {
  var btnBack = document.getElementById('wizBtnBack');
  var btnNext = document.getElementById('wizBtnNext');
  var total = 7;

  btnBack.style.display = _campWizardStep === 1 ? 'none' : 'block';

  if (_campWizardStep === 1) {
    btnNext.textContent = '✨ 点燃火种 →';
    btnNext.className = 'wiz-btn next primary';
  } else if (_campWizardStep === 6) {
    if (Object.keys(_campDraft.step6.confirmations).length >= _campDraft.step5.builders.length && _campDraft.step5.builders.length > 0) {
      btnNext.textContent = '🎊 启营仪式 →';
      btnNext.className = 'wiz-btn next primary';
    } else {
      btnNext.textContent = '📤 发送确认请求';
      btnNext.className = 'wiz-btn next';
    }
  } else if (_campWizardStep === 7) {
    btnBack.style.display = 'none';
    btnNext.textContent = '🏁 正式启动';
    btnNext.className = 'wiz-btn next primary';
  } else {
    btnNext.textContent = '下一步 →';
    btnNext.className = 'wiz-btn next';
  }
}

function generateCampSchedule() {
  var startDate = document.getElementById('cwStartDate').value;
  var days = parseInt(document.getElementById('cwDays').value) || 8;
  _campDraft.step2.startDate = startDate;
  _campDraft.step2.days = days;

  var schedule = [];
  var times = [{ time:'06:00', label:'上午' }, { time:'14:00', label:'下午' }, { time:'19:00', label:'晚上' }];
  times.forEach(function(t) {
    var cells = [];
    for (var i = 0; i < days; i++) cells.push('');
    schedule.push({ time: t.time, label: t.label, cells: cells });
  });
  _campDraft.step2.schedule = schedule;

  // 自动计算里程碑
  var ms = [
    { name:'宣传日', date:offsetDate(startDate, -14), rule:'开营-14天', locked:true },
    { name:'开营日', date:startDate, rule:'—', locked:true },
    { name:'展览日', date:offsetDate(startDate, 5), rule:'开营+5天', locked:true },
    { name:'结营日', date:offsetDate(startDate, days-1), rule:'开营+'+(days-1)+'天', locked:true }
  ];
  _campDraft.step2.milestones = ms;
  showToast('日程已生成！共 '+days+' 天', 'ok');
  renderWizardStep(2);
}

function offsetDate(dateStr, days) {
  var d = new Date(dateStr + 'T00:00:00');
  d.setDate(d.getDate() + days);
  return d.toISOString().slice(0, 10);
}

function renderMilestones(s) {
  if (!s.milestones || !s.milestones.length) return '';
  var h = '<div style="background:#fff;border:1px solid #d0d9ce;border-radius:10px;padding:12px;margin-top:8px"><div style="font-weight:700;font-size:.7rem;margin-bottom:8px">📍 里程碑节点</div>';
  s.milestones.forEach(function(m) {
    h += '<div style="display:flex;justify-content:space-between;padding:4px 0;font-size:.68rem;border-bottom:1px dotted #f0f0f0"><span>'+m.name+'</span><span style="color:#8a8a8a">'+m.date+'</span><span style="font-size:.6rem;color:#aaa">'+m.rule+'</span></div>';
  });
  h += '</div>';
  return h;
}
