**需要新增的 HTML 结构（在任务列表上方）：**
```html
<div class="activity-section" id="activitySection">
  <div class="activity-chips">
    <span class="chip active" onclick="filterActivity('all')">全部</span>
    <span class="chip" onclick="filterActivity('community')">🏘️社区</span>
    <span class="chip" onclick="filterActivity('camp')">🏕️营队</span>
    <span class="chip" onclick="filterActivity('personal')">👤个人</span>
  </div>
  <div id="activityList"></div>
</div>
```

**渲染函数：**
```javascript
var _activityFilter = 'all';
function renderActivity() {
  var entries = window.NT ? NT.getLedger({limit: 20}) : [];
  if (_activityFilter === 'community') entries = entries.filter(function(e){return e.type.indexOf('personal_')===0});
  if (_activityFilter === 'camp') entries = entries.filter(function(e){return e.type.indexOf('task_')===0});
  var h = '';
  entries.forEach(function(e) {
    var icon = e.type.indexOf('earn')>=0 ? '💰' : e.type.indexOf('spend')>=0 ? '💸' : '📝';
    h += '<div class="activity-item">'+icon+' '+e.reason+' · <b>'+e.amount+' NT</b> · '+e.timestamp.slice(5,10)+'</div>';
  });
  document.getElementById('activityList').innerHTML = h || '<div class="empty-state">暂无活动记录</div>';
}
function filterActivity(f) { _activityFilter = f; renderActivity(); }
```
