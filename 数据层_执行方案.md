# 数据层 — 执行方案（v2 · 审查修正版）

> 2026-07-18 · 审查修正：AppData 从「双写 NT」改为「包装 NT」，A1 从 4 表缩减为 2 表，补 invite_codes 到目标 Schema。
> **审查要点：** 单一真相源（NT 管钱，AppData 管 UI 数据）· 禁用双写 · 每步可验证

---

## 核心设计决策

```
NT 系统（金融真相源）              AppData（UI 数据 + 持久化）
─────────────────                ─────────────────────────
USERS[].ntBalance  ← 余额        users 表（bio, avatar, camp…）
LEDGER[]            ← 流水        tasks 表（note, submission, timeline…）
TASKS[]             ← escrow      items 表
verify()            ← 自检        统一 localStorage 持久化

AppData 不存钱。钱的真相源永远是 NT 系统。
AppData.ntBalance() → 调 NT.getUser(me).ntBalance
AppData.myTransactions() → 调 NT.getLedger({user: me})
AppData 不提供 addTransaction() —— 所有 NT 变动走 NT.earn()/NT.spend()
```

---

## 执行原则

1. **每一步可验证** — 改完跑 `NT.verify()` + 手动测试
2. **不改 UI** — 只换数据源，不碰 HTML/CSS
3. **向后兼容** — 旧变量（TASKS）逐步替换，过渡期用 getter 保护
4. **不双写** — AppData 永远不自己存交易/余额

## 步骤 A1：建 AppData 核心层（~30min）

### 新建文件：`js/app-data.js`

```javascript
// AppData — 统一数据层（包装 NT 系统，自管 UI 数据）
// 原则：NT 系统是金融唯一真相源。AppData 不存交易、不存余额。
window.AppData = {
  _currentUser: '砚仁',
  _data: {},

  // ══ 初始化 ══
  init: function() {
    var saved;
    try { saved = JSON.parse(localStorage.getItem('nt_app_v2')); } catch(e) { saved = null; }
    this._data = saved || {};
    if (!this._data.users)  this._data.users  = {};
    if (!this._data.tasks)  this._data.tasks  = {};
    if (!this._data.items)  this._data.items  = {};
    if (!this._data.camps)  this._data.camps  = {};
    if (!this._data.inviteCodes) this._data.inviteCodes = {};
    this._seedIfEmpty();
  },

  // ══ 当前用户 ══
  me: function(key) {
    var u = this._data.users[this._currentUser] || {};
    return key ? u[key] : u;
  },

  switchUser: function(name) {
    this._currentUser = name;
    this._save();
  },

  // ══ NT 余额/流水（读 NT 系统，不自己算） ══
  ntBalance: function() {
    if (!window.NT) return 0;
    var u = NT.getUser(this._currentUser);
    return u ? u.ntBalance : 0;
  },

  myTransactions: function(opts) {
    if (!window.NT) return [];
    return NT.getLedger({
      user: this._currentUser,
      limit: (opts && opts.limit) || 50
    });
  },

  // ══ 任务 ══
  myTasks: function() {
    var me = this._currentUser;
    var all = this._data.tasks;
    return Object.values(all).filter(function(t) {
      return t.publisher === me
        || (t.claimants || []).some(function(c) { return c.name === me; })
        || t.reviewer === me;
    });
  },

  taskMarket: function() {
    var all = this._data.tasks;
    return Object.values(all).filter(function(t) { return t.status !== 'draft'; });
  },

  // ══ 物品 ══
  myItems: function(filter) {
    var items = this._data.items[this._currentUser] || [];
    if (!filter || filter === 'all') return items;
    return items.filter(function(i) { return i.status === filter; });
  },

  // ══ 写入（只写 UI 数据，不碰 NT） ══
  setMe: function(key, val) {
    if (!this._data.users[this._currentUser]) this._data.users[this._currentUser] = {};
    this._data.users[this._currentUser][key] = val;
    this._save();
  },

  addTask: function(task) {
    this._data.tasks[task.name] = task;
    this._save();
  },

  updateTask: function(name, updates) {
    if (this._data.tasks[name]) Object.assign(this._data.tasks[name], updates);
    this._save();
  },

  deleteTask: function(name) {
    delete this._data.tasks[name];
    this._save();
  },

  addItem: function(item) {
    if (!this._data.items[this._currentUser]) this._data.items[this._currentUser] = [];
    this._data.items[this._currentUser].push(item);
    this._save();
  },

  updateItem: function(id, updates) {
    var items = this._data.items[this._currentUser] || [];
    var it = items.find(function(i) { return i.id === id; });
    if (it) Object.assign(it, updates);
    this._save();
  },

  // ══ 邀请码 ══
  validateInviteCode: function(code, campId) {
    var entry = this._data.inviteCodes[code];
    if (!entry) return { valid: false, reason: '邀请码不存在' };
    if (entry.usedBy) return { valid: false, reason: '邀请码已被使用' };
    if (entry.campId !== campId) return { valid: false, reason: '邀请码不匹配当前营队' };
    return { valid: true, entry: entry };
  },

  consumeInviteCode: function(code, userName) {
    if (!this._data.inviteCodes[code]) return false;
    this._data.inviteCodes[code].usedBy = userName;
    this._data.inviteCodes[code].usedAt = new Date().toISOString();
    this._save();
    return true;
  },

  // ══ 存盘 ══
  _timer: null,
  _save: function() {
    clearTimeout(this._timer);
    var self = this;
    this._timer = setTimeout(function() {
      try { localStorage.setItem('nt_app_v2', JSON.stringify(self._data)); } catch(e) {}
    }, 200);
  },

  // ══ 种子数据 ══
  _seedIfEmpty: function() {
    var self = this;
    if (Object.keys(this._data.users).length === 0) {
      this._data.users['砚仁'] = { name: '砚仁', type: 'admin', role: '总负责/NT审核', avatar_seed: 0, bio: '', camp: '第四期', location: '', created: '2026-06-06' };
      this._data.users['小云']  = { name: '小云',  type: 'builder', role: '', avatar_seed: 1, bio: '', camp: '第四期', location: '', created: '2026-07-01' };
      this._data.users['朝林']  = { name: '朝林',  type: 'builder', role: '', avatar_seed: 2, bio: '', camp: '第四期', location: '', created: '2026-07-01' };
      this._data.users['淑惠']  = { name: '淑惠',  type: 'visitor', role: '', avatar_seed: 3, bio: '', camp: '', location: '', created: '2026-07-01' };
      this._currentUser = '砚仁';
    }

    // 种子邀请码（§1 C2 需要）
    if (Object.keys(this._data.inviteCodes).length === 0) {
      this._data.inviteCodes['NT-A1B2C3D4'] = { campId: 'camp4', role: 'adventurer', createdBy: '砚仁', usedBy: null };
      this._data.inviteCodes['NT-E5F6G7H8'] = { campId: 'camp4', role: 'adventurer', createdBy: '砚仁', usedBy: null };
    }

    // NT 系统种子（金融数据，每次 init 都确保注册）
    if (window.NT) {
      if (!NT.getUser('砚仁')) NT.registerUser('砚仁', 500);
      if (!NT.getUser('小云'))  NT.registerUser('小云', 300);
      if (!NT.getUser('朝林'))  NT.registerUser('朝林', 200);
      if (!NT.getUser('淑惠'))  NT.registerUser('淑惠', 150);
    }

    this._save();
  },

  reset: function() {
    localStorage.removeItem('nt_app_v2');
    this._data = {};
    // 注意：NT 系统数据不在 localStorage 中（当前全内存），重置不会清 NT 余额。
    // 如需完全重置，刷新页面即可。
    this.init();
  }
};

// 自动初始化
AppData.init();
```

## 步骤 A3：迁移 NT 结算双写 + 物品系统（~20min）

### confirmSettle 改调 NT 系统

`confirmSettle()` L820-825，在 `AppData.updateTask(...)` 之后加：

```javascript
// 给完成者发 NT（从 NT 系统走，AppData 不自己记账）
var completed = (t.claimants || []).filter(function(c) { return c.status === 'completed'; });
completed.forEach(function(c) {
  NT.earn(c.name, t.nt, t.name + ' 结算', t.scope || 'personal');
  // 发布者支付 NT（从发布者余额扣）
  NT.spend(t.publisher || t.settler, t.nt, t.name + ' 支付', t.scope || 'personal');
});
```

**注意：** 此处修正了原 §0 R4 的语义方向——NT 现在正确地发给完成者，从发布者扣。`NT.spend()` 如果发布者余额不足会返回 null（nt-core.js 有余额检查）。
