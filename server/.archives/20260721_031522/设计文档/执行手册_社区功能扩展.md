#### 6.1-6.3 — 新建 `server/cron.py`

```python
import asyncio, json, logging
from datetime import datetime, timedelta
from database import async_session
from models import NTTask, CommunityPool
from sqlalchemy import select
from routes.nt import _get_pool, _add_ledger, _task_id

logger = logging.getLogger("cron")

async def tick_daily():
    """每日 00:05 触发"""
    today = datetime.utcnow().strftime("%Y-%m-%d")
    templates = json.load(open("config/periodic_tasks.json"))["templates"]
    async with async_session() as db:
        pool = await _get_pool(db)
        for tmpl in templates:
            if tmpl["period"] != "daily":
                continue
            # 幂等键
            idem_key = f"{tmpl['id']}:{today}"
            # 检查已存在
            existing = await db.execute(
                select(NTTask).where(NTTask.idempotency_key == idem_key)
            )
            if existing.scalar_one_or_none():
                continue
            # 池余额检查
            cost = tmpl["reward"] * tmpl["slots"]
            if pool.balance < cost:
                logger.warning(f"池余额不足，跳过 {tmpl['id']}")
                continue
            # 扣池 + 创建任务
            pool.balance -= cost
            pool.task_escrow += cost
            task = NTTask(
                id=_task_id(),
                poster="社区",
                title=tmpl["title"],
                reward=tmpl["reward"],
                slots=tmpl["slots"],
                scope=tmpl["scope"],
                category=tmpl["category"],
                status="进行中",
                is_system_generated=True,
                idempotency_key=idem_key,
                created_at=datetime.utcnow().isoformat()
            )
            db.add(task)
        await db.commit()

# 同样实现 tick_weekly()...
```
