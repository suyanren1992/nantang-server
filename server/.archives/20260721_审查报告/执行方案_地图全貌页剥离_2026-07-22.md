function renderContent(idx) {

### 步骤 3 · 新建 `openBuilding(idx)` / `closeBuilding()`（10 行）

```javascript
function openBuilding(idx) {
  var b = getBuildings()[idx];
  if (!b || b.id === 'info') return;
  
  var overlay = document.getElementById('overlayBuilding');
  overlay.classList.add('open');
  
  _q2('bldOverlayName').textContent = b.icon + ' ' + b.name;
  _q2('bldPhotoImg').style.background = b.photoBg;
  _q2('bldPhotoImg').innerHTML = '<div class="ph-fallback"><div class="ph-emoji">' + b.icon + '</div></div>';
  _q2('bldOverlayTitle').textContent = b.name;
  _q2('bldOverlayMeta').textContent = b.meta || '';
  
  _renderBuildingRooms(b);   // 从 renderContent 提取
  _renderBuildingFAB(b);     // 从 _updateFAB 提取
}

function closeBuilding() {
  document.getElementById('overlayBuilding').classList.remove('open');
  document.getElementById('overlayMap').classList.add('open');  // 回到全貌
}
```

`_q2` 是 `overlayBuilding` 内的元素查询（仿 `_q` 模式）：

```javascript
function _q2(id) { return document.getElementById(id); }
```
