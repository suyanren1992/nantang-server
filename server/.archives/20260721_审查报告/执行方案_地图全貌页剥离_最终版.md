# 地图全貌页剥离：最终执行方案

> 2026-07-22 · ~45min · 轮播保留 / 建筑独立 overlay / 互不干扰
> 设计参考：prototype/shared-kitchen.html（冰箱手风琴、物品行、底部固定操作栏）

---

## 需求

| # | 需求 | 方案如何满足 |
|---|------|-------------|
| 1 | 每个建筑独立 overlay，互不干扰 | 新建 overlayBuilding，一个 overlay 只渲染一个建筑 |
| 2 | 轮播保留（滑动、吸附、惯性） | 保留 renderCarousel + bcTrack + bcDots，不改交互逻辑 |
| 3 | 轮播仍是切换按键 | 轮播卡 onclick 从 goTo(idx) 改为 openBuilding(idx) |
| 4 | 全貌页是轮播里的一张卡 | idx=4（info）保留在 HARDCODED_BUILDINGS + 轮播中 |
| 5 | 建筑页内 ◀▶ 箭头保留 | overlayBuilding 的 photo 区保留左右箭头 |
| 6 | 刷新永远落在全貌页 | _initMap 不再恢复上次 idx，刷新时 force idx=4 |
| 7 | 参考厨房原型 | 房间物品行、FAB 固定底部、楼层手风琴 |
| 8 | 界面内容不动 | 全貌 6 板块 / 建筑照片+楼层+房间 / FAB — 全部保留 |

### 步骤 5 · 新建 `openBuilding(idx)`（app.js，+35 行）

在 `renderContent` 之后新增。建筑详情在独立 overlay 内完整渲染。

```javascript
var _bldFloor = 0, _bldSelectedRoom = null;

function openBuilding(idx) {
  var b = getBuildings()[idx];
  if (!b || b.id === 'info') return;

  // checkin_5 quest hook（从 goTo 迁移）
  if (b.id && window.AppData) {
    if (!AppData._data.visitedSpaces) AppData._data.visitedSpaces = [];
    if (AppData._data.visitedSpaces.indexOf(b.id) === -1) {
      AppData._data.visitedSpaces.push(b.id);
      if (AppData._data.visitedSpaces.length >= 5 && typeof _completeNewbieQuest === 'function')
        _completeNewbieQuest(CURRENT_USER, 'checkin_5');
      AppData._save();
    }
  }

  var ov = document.getElementById('overlayBuilding');
  ov.classList.add('open');
  if (!document.getElementById('overlayMap').classList.contains('open'))
    document.getElementById('overlayMap').classList.add('open');

  // 顶栏
  _qBld('bldName2').textContent = b.icon + ' ' + b.name;

  // 照片 + 左右箭头（保留建筑间切换）
  var ph = _qBld('bldPhoto2');
  ph.style.height = ''; ph.style.background = b.photoBg;
  ph.innerHTML = (b.photo ? '<img src="'+b.photo+'" class="ph-image" onerror="this.style.display=\'none\'">' : '') +
    '<div class="ph-fallback"><div class="ph-emoji">'+b.icon+'</div></div>' +
    '<button class="ph-arrow left" onclick="openBuilding('+(idx-1)+')">‹</button>' +
    '<button class="ph-arrow right" onclick="openBuilding('+(idx+1)+')">›</button>';

  _qBld('bldTitle2').textContent = b.name;
  _qBld('bldMeta2').textContent = b.meta || '';

  // 楼层 tab
  var fKeys = Object.keys(b.floors || {});
  var ft = '';
  fKeys.forEach(function(f,i){
    ft += '<button class="floor-tab'+(i===_bldFloor?' sel':'')+'" onclick="_setBldFloor('+i+','+idx+')">'+f+'</button>';
  });
  _qBld('floorTabs2').innerHTML = ft;
  _qBld('floorTabs2').style.display = ft ? 'flex' : 'none';

  // 房间网格 + FAB
  _renderBldRooms(b, idx);
  _renderBldFAB(b, idx);
  _bldFloor = 0; _bldSelectedRoom = null;

  // 同步轮播高亮
  if (typeof goTo === 'function') goTo(idx);
}
```

> `goTo(idx)` 在最后调用——这样轮播卡会吸附到当前建筑的位置高亮，同时触发 setMemberLocation 和 visitedSpaces。`openBuilding` 里已经做了 visitedSpaces，goTo 里的重复调用是幂等的，无害。

**`_renderBldRooms(b, idx)`** — 渲染房间网格，DOM scope 为 overlayBuilding：

```javascript
function _renderBldRooms(b, idx) {
  var rooms = _bldCurRooms(b);
  var isField = !!(b.plots && b.plots.length > 0);
  var rh = '';
  if (isField) {
    rh = typeof renderFieldPlots === 'function' ? renderFieldPlots(rooms) : '';
  } else if (rooms.length > 0) {
    rooms.forEach(function(r){
      var sel = _bldSelectedRoom === r.id ? ' selected' : '';
      rh += '<div class="room-card'+sel+'" onclick="_selectBldRoom(\''+r.id+'\','+idx+')"><div class="rc-dot" style="background:'+(r.status==='green'?'#5d8c52':r.status==='yellow'?'#c8892e':'#b84c38')+'"></div><div class="rc-icon">'+r.icon+'</div><div class="rc-name">'+r.name+'</div>'+(r.sub?'<div class="rc-sub">'+r.sub+'</div>':'')+'</div>';
    });
  } else {
    rh = '<div class="room-empty">📍 开放空间</div>';
  }
  _qBld('roomsGrid2').innerHTML = rh;
  _qBld('roomsGrid2').style.display = '';

  // 选中房间时显示物品详情（复用 overlayMap 内的 itemsOverlay）
  if (_bldSelectedRoom) {
    var room = rooms.find(function(r){return r.id===_bldSelectedRoom;});
    if (room) {
      document.getElementById('itemsOverlay').classList.add('show');
      _q('ioTitle').textContent = room.icon + ' ' + room.name;
      _q('ioBody').innerHTML = isField && typeof buildFieldDetail === 'function' ? buildFieldDetail(room) : (typeof buildRoomDetail === 'function' ? buildRoomDetail(room) : '');
    }
  } else {
    document.getElementById('itemsOverlay').classList.remove('show');
  }
}

function _selectBldRoom(roomId, idx) {
  _bldSelectedRoom = (_bldSelectedRoom === roomId ? null : roomId);
  var b = getBuildings()[idx];
  if (b) _renderBldRooms(b, idx);
}
```

**`_bldCurRooms(b)`** — 不依赖 `currentIdx` 的当前楼层房间获取：

function _bldCurRooms(b) {
  if (!b) return [];
  var fKeys = Object.keys(b.floors || {});
  var floorKey = fKeys[_bldFloor] || '';
  if (b.id === 'field') return typeof getPlots === 'function' ? getPlots() : [];
  if (b.floors && b.floors[floorKey]) return b.floors[floorKey];
  return [];
}
